"""
Risk Stratification Logic for RiskLens-AD
-----------------------------------------

Defines how predicted probabilities are converted into
clinically meaningful risk bands.
"""

def risk_band(probability):
    """
    Map predicted probability to risk category.
    
    Parameters
    ----------
    probability : float
        Model predicted probability of dementia risk.

    Returns
    -------
    str
        Risk category label.
    """
    
    if probability < 0.35:
        return "Low Risk (Confident)"
    elif probability > 0.65:
        return "High Risk (Confident)"
    else:
        return "Uncertain / Needs Monitoring"
